1) . const [cartItems, setCartItems] = useState({});

  const addToCart = (itemId) => {
    if(!cartItems[itemId]) {
      setCartItems(prev => ({
        ...prev,
        [itemId]: 1
      }));
    } else {
      setCartItems(prev => ({
        ...prev,
        [itemId]: prev[itemId] + 1
      }));
    }
  }
